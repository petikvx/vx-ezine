function lerfissetor(_seg,_ofs:word;trilha,setor,cab,unid:byte):boolean;
var                                                      
        tent : byte;                                                    
        carry : byte;                                                   
begin                                                    
        tent := 0;                                                    
        repeat                                                        
                carry := 0;                                                   
                asm                                                  
                   mov ax,201h                                                 
                   mov es,_seg                                                  
                   mov bx,_ofs                                                  
                   mov ch,trilha                                                  
                   mov cl,setor                                                 
                   mov dh,cab                                                  
                   mov dl,unid                                                 
                   int 13h                                                  
                   jnc @@1                                                  
                   mov carry,1                                                  
                   @@1:                                                    
                   end;                                                  
                tent := tent +1;                                                  
        until (tent>5) or (carry=0);                                             
        if (tent >5) and (carry<>0)then                                           
                lerfissetor := false else lerfissetor := true;                  
        end;
