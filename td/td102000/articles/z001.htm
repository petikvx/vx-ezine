написать минимальную подпрограмму обмена в AX'е битов номер DL и DH

; input: (E)AX,DH,DL
; action: swap (E)AX.bit[DL], (E)AX.bit[DH]
; size: 17 bytes
; code: 16/32-bit

8A DE           bitswap:mov     bl, dh
B7 5A                   mov     bh, 'Z'         ; (c)
0F BB D0        cycle:  btc     (e)ax, (e)dx
12 FF                   adc     bh, bh
0F BB D8                btc     (e)ax, (e)bx
12 FF                   adc     bh, bh
7B F4                   jnp     cycle
C3                      retn
